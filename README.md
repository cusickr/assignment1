#!/bin/bash

# System Information
USERNAME=$(whoami)
CURRENT_DATETIME=$(date '+%Y-%m-%d %H:%M:%S')
HOSTNAME=$(hostname)
OS="${NAME} ${VERSION}"
UPTIME=$(uptime -p)

# System Report Output
cat <<EOF

System Report generated by $USERNAME, $CURRENT_DATETIME

System Information
------------------
Hostname: $HOSTNAME
OS: $OS
Uptime: $UPTIME
